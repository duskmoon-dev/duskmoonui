# Checkbox

Checkboxes allow users to select one or more items from a set or toggle a single option. @duskmoon-dev/core provides a complete set of Material Design 3 checkbox variants with support for checked, indeterminate, and disabled states.

## Basic Usage

```html
<label class="checkbox">
  <input type="checkbox" class="checkbox-input" />
  <span class="checkbox-box">
    <svg class="checkbox-checkmark" fill="none" viewBox="0 0 24 24" stroke="currentColor">
      <path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M5 13l4 4L19 7" />
    </svg>
  </span>
</label>
```

```html
    </label>

    <label class="checkbox">
      <input type="checkbox" class="checkbox-input" class="child-checkbox" />
      <span class="checkbox-box">
        <svg class="checkbox-checkmark" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M5 13l4 4L19 7" />
        </svg>
      </span>
      <span class="checkbox-label">Option 2</span>
    </label>

    <label class="checkbox">
      <input type="checkbox" class="checkbox-input" class="child-checkbox" />
      <span class="checkbox-box">
        <svg class="checkbox-checkmark" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M5 13l4 4L19 7" />
        </svg>
      </span>
      <span class="checkbox-label">Option 3</span>
    </label>
  </div>
</div>

<script>
  const parent = document.getElementById('select-all');
  const children = document.querySelectorAll('.child-checkbox');

  parent.addEventListener('change', function() {
    children.forEach(child => child.checked = this.checked);
    this.indeterminate = false;
  });

  children.forEach(child => {
    child.addEventListener('change', function() {
      const checkedCount = Array.from(children).filter(c => c.checked).length;
      parent.checked = checkedCount === children.length;
      parent.indeterminate = checkedCount > 0 && checkedCount < children.length;
    });
  });
</script>
```

### Accessibility

- Always use semantic `<label>` elements to associate labels with checkboxes
- Provide clear, descriptive labels
- Use `fieldset` and `legend` for checkbox groups
- Ensure sufficient color contrast
- Support keyboard navigation (built-in with native checkbox)
- Use `aria-describedby` for helper text

```html
<div>
  <label class="checkbox">
    <input
      type="checkbox"
      class="checkbox-input"
      aria-describedby="newsletter-help"
    />
    <span class="checkbox-box">
      <svg class="checkbox-checkmark" fill="none" viewBox="0 0 24 24" stroke="currentColor">
        <path stroke-linecap="round" stroke-linejoin="round" stroke-width="3" d="M5 13l4 4L19 7" />
      </svg>
    </span>
    <span class="checkbox-label">Subscribe to newsletter</span>
  </label>
  <span class="checkbox-helper" id="newsletter-help">
    We'll send you weekly updates. You can unsubscribe anytime.
  </span>
</div>
```

## Framework Examples

### React

```tsx
interface CheckboxProps {
  variant?: 'primary' | 'secondary' | 'tertiary';
  size?: 'sm' | 'md' | 'lg';
  label?: string;
  helperText?: string;
  error?: boolean;
  disabled?: boolean;
  checked?: boolean;
  indeterminate?: boolean;
  onChange?: (checked: boolean) => void;
}

export function Checkbox({
  variant = 'primary',
  size = 'md',
  label,
  helperText,
  error,
  disabled,
  checked,
  indeterminate,
  onChange,
}: CheckboxProps) {
  const ref = useRef<HTMLInputElement>(null);

  useEffect(() => {
    if (ref.current) {
      ref.current.indeterminate = indeterminate ?? false;
    }
  }, [indeterminate]);

  const boxClasses = [
    'checkbox-box',
    variant !== 'primary' && `checkbox-${variant}`,
  ].filter(Boolean).join(' ');

  const containerClasses = [
    'checkbox',
    size !== 'md' && `checkbox-${size}`,
    error && 'checkbox-error',
  ].filter(Boolean).join(' ');

  return (
    <div>
      <label className={containerClasses}>
        <input
          ref={ref}
          type="checkbox"
          className="checkbox-input"
          checked={checked}
          disabled={disabled}
          onChange={(e) => onChange?.(e.target.checked)}
        />
        <span className={boxClasses}>
          <svg className="checkbox-checkmark" fill="none" viewBox="0 0 24 24" stroke="currentColor">
            <path
              strokeLinecap="round"
              strokeLinejoin="round"
              strokeWidth="3"
              d={indeterminate ? "M5 12h14" : "M5 13l4 4L19 7"}
            />
          </svg>
        </span>
        {label && <span className="checkbox-label">{label}</span>}
      </label>
      {helperText && <span className="checkbox-helper">{helperText}</span>}
    </div>
  );
}

// Usage
<Checkbox
  label="Accept terms and conditions"
  helperText="You must accept to continue"
  error={!accepted}
  checked={accepted}
  onChange={setAccepted}
/>
```

### Vue

```vue
<template>
  <div>
    <label :class="containerClasses">
      <input
        ref="input"
        type="checkbox"
        class="checkbox-input"
        :checked="checked"
        :disabled="disabled"
        @change="$emit('update:checked', $event.target.checked)"
      />
      <span :class="boxClasses">
        <svg class="checkbox-checkmark" fill="none" viewBox="0 0 24 24" stroke="currentColor">
          <path
            stroke-linecap="round"
            stroke-linejoin="round"
            stroke-width="3"
            :d="indeterminate ? 'M5 12h14' : 'M5 13l4 4L19 7'"
          />
        </svg>
      </span>
      <span v-if="label" class="checkbox-label">{{ label }}</span>
    </label>
    <span v-if="helperText" class="checkbox-helper">{{ helperText }}</span>
  </div>
</template>

<script setup>
import { ref, computed, watch, onMounted } from 'vue'

const props = defineProps({
  variant: {
    type: String,
    default: 'primary'
  },
  size: {
    type: String,
    default: 'md'
  },
  label: String,
  helperText: String,
  error: Boolean,
  disabled: Boolean,
  checked: Boolean,
  indeterminate: Boolean
})

defineEmits(['update:checked'])

const input = ref(null)

const containerClasses = computed(() => [
  'checkbox',
  props.size !== 'md' && `checkbox-${props.size}`,
  props.error && 'checkbox-error'
].filter(Boolean).join(' '))

const boxClasses = computed(() => [
  'checkbox-box',
  props.variant !== 'primary' && `checkbox-${props.variant}`
].filter(Boolean).join(' '))

watch(() => props.indeterminate, (value) => {
  if (input.value) {
    input.value.indeterminate = value ?? false
  }
})

onMounted(() => {
  if (input.value && props.indeterminate) {
    input.value.indeterminate = true
  }
})
</script>

<!-- Usage -->
<Checkbox
  v-model:checked="accepted"
  label="Accept terms and conditions"
  helper-text="You must accept to continue"
  :error="!accepted"
/>
```

## API Reference

### Class Names

| Class | Description |
|-------|-------------|
| `.checkbox` | Base checkbox container (required) |
| `.checkbox-input` | Hidden native checkbox input (required) |
| `.checkbox-box` | Visual checkbox box (required) |
| `.checkbox-checkmark` | Checkmark icon (required) |
| `.checkbox-primary` | Primary color variant (default) |
| `.checkbox-secondary` | Secondary color variant |
| `.checkbox-tertiary` | Tertiary color variant |
| `.checkbox-sm` | Small size (16px) |
| `.checkbox-md` | Medium size (20px, default) |
| `.checkbox-lg` | Large size (24px) |
| `.checkbox-label` | Label text |
| `.checkbox-label-left` | Position label on the left |
| `.checkbox-helper` | Helper/error text below checkbox |
| `.checkbox-error` | Error state styling |
| `.checkbox-loading` | Loading animation state |

### HTML Attributes

| Attribute | Description |
|-----------|-------------|
| `checked` | Sets the checkbox to checked state |
| `disabled` | Disables the checkbox interaction |
| `indeterminate` | Sets indeterminate state (via JavaScript) |
| `required` | Marks the checkbox as required for form validation |
| `aria-describedby` | Associates helper text with the checkbox |

### States

- **Default**: Unchecked state with outline border
- **Hover**: Slightly darkened background on hover
- **Focus**: Visible focus ring when navigating with keyboard
- **Checked**: Filled background with checkmark visible
- **Indeterminate**: Filled background with dash/minus icon
- **Disabled**: Reduced opacity and cursor-not-allowed
- **Error**: Red border and error text color
- **Loading**: Pulsing animation effect
